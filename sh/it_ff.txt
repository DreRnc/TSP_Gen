# Common flags for both commands
common_flags="-t -P 5000 -p 4096 -g 10"

# Iterate over the values of $i for luxembourg
mkdir results/luxembourg
for ((i = 2; i <= 64; i *= 2)) 
do 
    # Concatenate common flags with the changing flags
    flags="$common_flags -w $i -f "results/luxembourg/time_ff.txt" -d "data/luxembourg.tsp""
    
    # Execute the command with the concatenated flags
    echo "Executing ./bin/tspg_par_ff $flags"
    ./bin/tspg_par_ff $flags
done
