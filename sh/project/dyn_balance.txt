# Common flags for both sets of commands
common_flags="-t -P 16384 -p 8192 -g 10 -d data/djibouti.tsp"

rm -r results/djibouti
mkdir -p results/djibouti

for ((j = 2; j <= 64; j *= 2)) 
do  
    echo "Executing with chunk size $j"
    ./bin/tspg_seq $common_flags -f results/djibouti/time_seq.txt

    output_file="results/djibouti/time_dyn_cs_${j}.txt"
    
    # Iterate over the values of $i for ./bin/tspg_par_native_dyn
    for ((i = 1; i <= 64; i *= 2)) 
    do 
        # Concatenate common flags with the changing flags
        flags="$common_flags -w $i -ds $j -f $output_file"
        
        # Execute the command with the concatenated flags
        ./bin/tspg_par_native_dyn $flags
    done
done
